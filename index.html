import fitz # PyMuPDF # Open the PDF file pdf_path = 'path_to_your_pdf_file.pdf' pdf_document = fitz.open(pdf_path) # Initialize an empty string to store text content text_content = '' # Iterate through each page and extract text for page_num in range(len(pdf_document)): page = pdf_document.load_page(page_num) text_content += page.get_text() # Close the PDF document pdf_document.close() # Format the extracted text into HTML html_content = f''' PDF to HTML
{text_content}
''' # Save as index.html html_file_path = 'index.html' with open(html_file_path, 'w', encoding='utf-8') as html_file: html_file.write(html_content) print(f'HTML file saved at: {html_file_path}')
